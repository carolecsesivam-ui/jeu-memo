  <script>

    // Liste des modèles Toyota (6 colonnes → 24 cartes = 12 paires)

    const models = [

      "Yaris",

      "Corolla",

      "Prius",

      "RAV4",

      "Hilux",

      "C-HR",

      "Camry",

      "RAV4",

      "Hilux",

      "Yaris",

      "Corolla",

      "Prius",

      "C-HR",

      "Camry",

      "Land Cruiser",

      "Aygo",

      "GR86",

      "Supra",

      "Tacoma",

      "Land Cruiser",

      "Aygo",

      "GR86",

      "Supra",

      "Tacoma"

    ];

 

    const gameDiv = document.getElementById("game");

    let cards = [];

    let firstCard = null;

    let secondCard = null;

    let canFlip = true;

 

    function initGame() {

      // Mélange les cartes

      const shuffled = [...models].sort(() => Math.random() - 0.5);

 

      // Crée les cartes

      shuffled.forEach((modelName, index) => {

        const card = document.createElement("div");

        card.className = "card";

        card.dataset.name = modelName;

 

        const front = document.createElement("div");

        front.className = "back";

        front.innerHTML = `<span>?</span>`;

 

        const back = document.createElement("img");

        back.src = `https://source.unsplash.com/200x260/?toyota,cars`;

        back.alt = "Toyota " + modelName;

 

        card.appendChild(front);

        card.appendChild(back);

        card.addEventListener("click", flipCard);

 

        gameDiv.appendChild(card);

        cards.push(card);

      });

    }

 

    function flipCard() {

      if (!canFlip || this === firstCard) return;

 

      if (!firstCard) {

        // 1ère carte

        firstCard = this;

        this.style.transform = "rotateY(180deg)";

        this.children[0].style.display = "none";

        return;

      }

 

      // 2ème carte

      secondCard = this;

      canFlip = false;

      this.style.transform = "rotateY(180deg)";

      this.children[0].style.display = "none";

 

      if (firstCard.dataset.name === secondCard.dataset.name) {

        // Bonne paire

        setTimeout(() => {

          firstCard.style.visibility = "hidden";

          secondCard.style.visibility = "hidden";

          firstCard = null;

          secondCard = null;

          canFlip = true;

        }, 800);

      } else {

        // Mauvaise paire

        setTimeout(() => {

          firstCard.style.transform = "rotateY(0)";

          firstCard.children[0].style.display = "flex";

          secondCard.style.transform = "rotateY(0)";

          secondCard.children[0].style.display = "flex";

          firstCard = null;

          secondCard = null;

          canFlip = true;

        }, 1200);

      }

    }

 

    // Ajoute un peu de style 3D basique

    const style = document.createElement("style");

    style.textContent = ".card { transform-style: preserve-3d; }";

    document.head.appendChild(style);

 

    // Lancement

    initGame();

  </script>
